def time_of_trip(datum, city):
    """
    
    
    
    
    
    
     datetime 
    
     https://docs.python.org/3/library/datetime.html#strftime-and-strptime-behavior
    """
    
    # 
    weekdays = ['Monday','Tuesday','Wednesday','Thursday','Friday','Saturday','Sunday']
    if city == 'NYC':
        dtime = datetime.timetuple(datetime.strptime(datum['starttime'],'%m/%d/%Y %H:%M:%S'))
        
    elif city =='Chicago':
        dtime = datetime.timetuple(datetime.strptime(datum['stoptime'],'%m/%d/%Y %H:%M'))
        
    else:
        dtime = datetime.timetuple(datetime.strptime(datum['Start date'],'%m/%d/%Y %H:%M'))
    month,hour,day_of_week = dtime[1],dtime[3],weekdays[dtime[6]]
    return (month, hour, day_of_week)

tests = {'NYC': (1, 0, 'Friday'),
         'Chicago': (3, 23, 'Thursday'),
         'Washington': (3, 22, 'Thursday')}

for city in tests:
    assert time_of_trip(example_trips[city], city) == tests[city]

def condense_data(in_file, out_file, city):
    """
    
    
    
    
    
    """
    
    with open(out_file, 'w') as f_out, open(in_file, 'r') as f_in:
        #  csv DictWriter 
        #  "fieldnames" 
        out_colnames = ['duration', 'month', 'hour', 'day_of_week', 'user_type']        
        trip_writer = csv.DictWriter(f_out, fieldnames = out_colnames)
        trip_writer.writeheader()
        
        ##  csv DictReader ##
        trip_reader = csv.DictReader(f_in)

        # 
        for row in trip_reader:
            # 
            new_point = {}

            ## ##
            new_point['duration'] = duration_in_mins(row,city)
            new_point['month'],new_point['hour'],new_point['day_of_week'] = time_of_trip(row,city)
            new_point['user_type'] = type_of_user(row,city)

            ##  new_point  ##
            ##  DictWriter         ##
            

            ## ##
            ##  https://docs.python.org/3/library/csv.html#writer-objects ##
            trip_writer.writerow(new_point)

def number_of_trips(filename):
    """
    
    
    """
    with open(filename, 'r') as f_in:
        #  csv reader 
        reader = csv.DictReader(f_in)
        
        # 
        n_subscribers = 0
        n_customers = 0
        n_none = 0
        
        # 
        for row in reader:
            if row['user_type'] == 'Subscriber':
                n_subscribers += 1
            elif row['user_type'] == 'Customer':
                n_customers += 1
            else:
                n_none += 1
        
        # 
        n_total = n_subscribers + n_customers + n_none
        p_subscribers = n_subscribers/n_total
        p_customers = n_customers/n_total
        
        # 
        return(n_subscribers, n_customers, n_total, '%.2f%%'%(p_subscribers*100), '%.2f%%'%(p_customers*100))

def average(list):
    total = 0
    for number in list:
        total += number
    return total / len(list)

def month_of_trips(filename):
    ''' ''' 
    with open(filename,'r') as f_in:
        reader = csv.DictReader(f_in)
        
        ## list1,list2,list3
        list1,list2,list3 = [],[],[]
        list4= range(1,13)
        month_users,month_subscribers,month_customers,rate = {},{},{},{}
        for row in reader:
            list1.append(int(row['month']))
            if row['user_type'] == 'Subscriber':
                list2.append(int(row['month']))
            elif row['user_type'] == 'Customer':
                list3.append(int(row['month']))
        for number in list4:
            month_users[number] = list1.count(number)
            month_subscribers[number] = list2.count(number)
            month_customers[number] = list3.count(number)
            rate[number] = '%.2f%%'%(month_customers[number]*100/month_subscribers[number])
        return month_users,month_subscribers,month_customers,rate
